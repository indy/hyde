Not

CHIP Not {

    IN  in;
    OUT out;

    PARTS:
    Nand(a=in,b=in,out=out);
}

And

CHIP And {

    IN  a, b;
    OUT out;

    PARTS:
    Nand(a=a,b=b,out=foo);
    Not(in=foo,out=out);
}

Or

CHIP Or {

    IN  a, b;
    OUT out;

    PARTS:
    Nand(a=a,b=a,out=foo);
    Nand(a=b,b=b,out=bar);
    Nand(a=foo,b=bar,out=out);
}

Xor

CHIP Xor {

    IN  a, b;
    OUT out;

    PARTS:
    Not(in=a,out=not1);
    Not(in=b,out=not2);
    Nand(a=not1,b=not2,out=foo);
    Nand(a=a,b=b,out=bar);
    And(a=foo,b=bar,out=out);
}

Mux

CHIP Mux {

    IN  a, b, sel;
    OUT out;

    PARTS:
    Not(in=sel,out=notsel);
    And(a=notsel,b=a,out=foo);
    And(a=sel,b=b,out=bar);
    Or(a=foo,b=bar,out=out);
}

DMux

CHIP DMux {

    IN  in, sel;
    OUT a, b;

    PARTS:
    Not(in=sel,out=notsel);
    And(a=notsel,b=in,out=a);
    And(a=sel,b=in,out=b);
}

    PARTS:

    Mux4Way16(a=e,b=f,c=g,d=h,sel=sel[0..1],out=efgh);
    Mux16(a=abcd,b=efgh,sel=sel[2],out=out);
}

DMux4Way

CHIP DMux4Way {

    IN  in, sel[2];
    OUT a, b, c, d;

    PARTS:
    DMux(in=in,sel=sel[0],a=ac,b=bd);
    DMux(in=in,sel=sel[1],a=ab,b=cd);
    And(a=ac,b=ab,out=a);
    And(a=bd,b=ab,out=b);
    And(a=ac,b=cd,out=c);
    And(a=bd,b=cd,out=d);
}

CHIP DMux8Way {

    DMux4Way(in=abcd,sel=sel[0..1],a=a,b=b,c=c,d=d);
    DMux4Way(in=efgh,sel=sel[0..1],a=e,b=f,c=g,d=h);
}
